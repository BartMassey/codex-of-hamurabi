To run a year of hmrabi:
  -- Add babies / immigrants
  immigrants ← floor((random 1..5) times (20 times acreage + stored) / (100 times P) + 1)
  pop ← pop + immigrants
  record immigrants
  record pop
  -- Check for plague
  plagued ← 0
  if 15% chance
    plagued ← ceil(pop / 2)
    pop -= plagued
  record plagued
  -- Update acreage
  lprice ← randint(10) + 17
  input purch
  lbushels ← lprice times purch
  assert acreage ≥ -purch ∧ lbushels ≤ stored
  acreage += purch
  stored -= lbushels
  record lprice
  record purch
  record acreage
  -- Feed people
  input feed
  assert feed ≥ 0 ∧ feed ≤ stored
  stored -= feed
  record feed
  -- Plant crops
  input plant
  assert plant ≥ 0
  assert floor(plant) ≤ 2 times stored
  assert plant ≤ 10 times pop
  stored -= floor(plant)
  record plant
  -- Harvest crops
  yield ← random 1..5
  harvest ← plant times yield
  record yield
  record harvest
  -- Feed rats
  ratchance ← random 1..5
  eaten ← 0
  if ratchance is odd
    eaten ← floor(stored / ratchance)
  stored -= eaten
  record eaten
  -- Check for sufficiently fed
  dead ← population - fed
  if dead > 45% of population
    failed ← true
  record dead
  record stored
